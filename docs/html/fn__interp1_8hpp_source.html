22template<typename eT>
23inline
24void
25interp1_helper_nearest(const Mat<eT>& XG, const Mat<eT>& YG, const Mat<eT>& XI, Mat<eT>& YI, const eT extrap_val)
26 {
27 arma_extra_debug_sigprint();
28
29 const eT XG_min = XG.min();
30 const eT XG_max = XG.max();
31
32 YI.copy_size(XI);
33
34 const eT* XG_mem = XG.memptr();
35 const eT* YG_mem = YG.memptr();
36 const eT* XI_mem = XI.memptr();
37 eT* YI_mem = YI.memptr();
38
39 const uword NG = XG.n_elem;
40 const uword NI = XI.n_elem;
41
42 uword best_j = 0;
43
44 for(uword i=0; i<NI; ++i)
45 {
46 eT best_err = Datum<eT>::inf;
47
48 const eT XI_val = XI_mem[i];
49
50 if((XI_val < XG_min) || (XI_val > XG_max))
51 {
52 YI_mem[i] = extrap_val;
53 }
54 else
55 {
56 // XG and XI are guaranteed to be sorted in ascending manner,
57 // so start searching XG from last known optimum position
58
59 for(uword j=best_j; j<NG; ++j)
60 {
61 const eT tmp = XG_mem[j] - XI_val;
62 const eT err = (tmp >= eT(0)) ? tmp : -tmp;
63
64 if(err >= best_err)
65 {
66 // error is going up, so we have found the optimum position
67 break;
68 }
69 else
70 {
71 best_err = err;
72 best_j = j; // remember the optimum position
73 }
74 }
75
76 YI_mem[i] = YG_mem[best_j];
77 }
78 }
79 }

83template<typename eT>
84inline
85void
86interp1_helper_linear(const Mat<eT>& XG, const Mat<eT>& YG, const Mat<eT>& XI, Mat<eT>& YI, const eT extrap_val)
87 {
88 arma_extra_debug_sigprint();
89
90 const eT XG_min = XG.min();
91 const eT XG_max = XG.max();
92
93 YI.copy_size(XI);
94
95 const eT* XG_mem = XG.memptr();
96 const eT* YG_mem = YG.memptr();
97 const eT* XI_mem = XI.memptr();
98 eT* YI_mem = YI.memptr();
99
100 const uword NG = XG.n_elem;
101 const uword NI = XI.n_elem;
102
103 uword a_best_j = 0;
104 uword b_best_j = 0;
105
106 for(uword i=0; i<NI; ++i)
107 {
108 const eT XI_val = XI_mem[i];
109
110 if((XI_val < XG_min) || (XI_val > XG_max))
111 {
112 YI_mem[i] = extrap_val;
113 }
114 else
115 {
116 // XG and XI are guaranteed to be sorted in ascending manner,
117 // so start searching XG from last known optimum position
118
119 eT a_best_err = Datum<eT>::inf;
120 eT b_best_err = Datum<eT>::inf;
121
122 for(uword j=a_best_j; j<NG; ++j)
123 {
124 const eT tmp = XG_mem[j] - XI_val;
125 const eT err = (tmp >= eT(0)) ? tmp : -tmp;
126
127 if(err >= a_best_err)
128 {
129 break;
130 }
131 else
132 {
133 a_best_err = err;
134 a_best_j = j;
135 }
136 }
137
138 if( (XG_mem[a_best_j] - XI_val) <= eT(0) )
139 {
140 // a_best_j is to the left of the interpolated position
141
142 b_best_j = ( (a_best_j+1) < NG) ? (a_best_j+1) : a_best_j;
143 }
144 else
145 {
146 // a_best_j is to the right of the interpolated position
147
148 b_best_j = (a_best_j >= 1) ? (a_best_j-1) : a_best_j;
149 }
150
151 b_best_err = std::abs( XG_mem[b_best_j] - XI_val );
152
153 if(a_best_j > b_best_j)
154 {
155 std::swap(a_best_j, b_best_j );
156 std::swap(a_best_err, b_best_err);
157 }
158
159 const eT weight = (a_best_err > eT(0)) ? (a_best_err / (a_best_err + b_best_err)) : eT(0);
160
161 YI_mem[i] = (eT(1) - weight)*YG_mem[a_best_j] + (weight)*YG_mem[b_best_j];
162 }
163 }
164 }

168template<typename eT>
169inline
170void
171interp1_helper(const Mat<eT>& X, const Mat<eT>& Y, const Mat<eT>& XI, Mat<eT>& YI, const uword sig, const eT extrap_val)
172 {
173 arma_extra_debug_sigprint();
174
175 arma_debug_check( ((X.is_vec() == false) || (Y.is_vec() == false) || (XI.is_vec() == false)), "interp1(): currently only vectors are supported" );
176
177 arma_debug_check( (X.n_elem != Y.n_elem), "interp1(): X and Y must have the same number of elements" );
178
179 arma_debug_check( (X.n_elem < 2), "interp1(): X must have at least two unique elements" );
180
181 // sig = 10: nearest neighbour
182 // sig = 11: nearest neighbour, assume monotonic increase in X and XI
183 //
184 // sig = 20: linear
185 // sig = 21: linear, assume monotonic increase in X and XI
186
187 if(sig == 11) { interp1_helper_nearest(X, Y, XI, YI, extrap_val); return; }
188 if(sig == 21) { interp1_helper_linear (X, Y, XI, YI, extrap_val); return; }
189
190 uvec X_indices;
191
192 try { X_indices = find_unique(X,false); } catch(...) { }
193
194 // NOTE: find_unique(X,false) provides indices of elements sorted in ascending order
195 // NOTE: find_unique(X,false) will reset X_indices if X has NaN
196
197 const uword N_subset = X_indices.n_elem;
198
199 arma_debug_check( (N_subset < 2), "interp1(): X must have at least two unique elements" );
200
201 Mat<eT> X_sanitised(N_subset, 1, arma_nozeros_indicator());
202 Mat<eT> Y_sanitised(N_subset, 1, arma_nozeros_indicator());
203
204 eT* X_sanitised_mem = X_sanitised.memptr();
205 eT* Y_sanitised_mem = Y_sanitised.memptr();
206
207 const eT* X_mem = X.memptr();
208 const eT* Y_mem = Y.memptr();
209
210 const uword* X_indices_mem = X_indices.memptr();
211
212 for(uword i=0; i<N_subset; ++i)
213 {
214 const uword j = X_indices_mem[i];
215
216 X_sanitised_mem[i] = X_mem[j];
217 Y_sanitised_mem[i] = Y_mem[j];
218 }
219
220
221 Mat<eT> XI_tmp;
222 uvec XI_indices;
223
224 const bool XI_is_sorted = XI.is_sorted();
225
226 if(XI_is_sorted == false)
227 {
228 XI_indices = sort_index(XI);
229
230 const uword N = XI.n_elem;
231
232 XI_tmp.copy_size(XI);
233
234 const uword* XI_indices_mem = XI_indices.memptr();
235
236 const eT* XI_mem = XI.memptr();
237 eT* XI_tmp_mem = XI_tmp.memptr();
238
239 for(uword i=0; i<N; ++i)
240 {
241 XI_tmp_mem[i] = XI_mem[ XI_indices_mem[i] ];
242 }
243 }
244
245 const Mat<eT>& XI_sorted = (XI_is_sorted) ? XI : XI_tmp;
246
247
248 if(sig == 10) { interp1_helper_nearest(X_sanitised, Y_sanitised, XI_sorted, YI, extrap_val); }
249 else if(sig == 20) { interp1_helper_linear (X_sanitised, Y_sanitised, XI_sorted, YI, extrap_val); }
250
251
252 if( (XI_is_sorted == false) && (YI.n_elem > 0) )
253 {
254 Mat<eT> YI_unsorted;
255
256 YI_unsorted.copy_size(YI);
257
258 const eT* YI_mem = YI.memptr();
259 eT* YI_unsorted_mem = YI_unsorted.memptr();
260
261 const uword N = XI_sorted.n_elem;
262 const uword* XI_indices_mem = XI_indices.memptr();
263
264 for(uword i=0; i<N; ++i)
265 {
266 YI_unsorted_mem[ XI_indices_mem[i] ] = YI_mem[i];
267 }
268
269 YI.steal_mem(YI_unsorted);
270 }
271 }

275template<typename T1, typename T2, typename T3>
276inline
277typename
278enable_if2
279 <
280 is_real<typename T1::elem_type>::value,
281 void
282 >::result
283interp1
284 (
285 const Base<typename T1::elem_type, T1>& X,
286 const Base<typename T1::elem_type, T2>& Y,
287 const Base<typename T1::elem_type, T3>& XI,
288 Mat<typename T1::elem_type>& YI,
289 const char* method = "linear",
290 const typename T1::elem_type extrap_val = Datum<typename T1::elem_type>::nan
291 )
292 {
293 arma_extra_debug_sigprint();
294
295 typedef typename T1::elem_type eT;
296
297 uword sig = 0;
298
299 if(method != nullptr)
300 if(method[0] != char(0))
301 if(method[1] != char(0))
302 {
303 const char c1 = method[0];
304 const char c2 = method[1];
305
306 if(c1 == 'n') { sig = 10; } // nearest neighbour
307 else if(c1 == 'l') { sig = 20; } // linear
308 else
309 {
310 if( (c1 == '*') && (c2 == 'n') ) { sig = 11; } // nearest neighour, assume monotonic increase in X and XI
311 if( (c1 == '*') && (c2 == 'l') ) { sig = 21; } // linear, assume monotonic increase in X and XI
312 }
313 }
314
315 arma_debug_check( (sig == 0), "interp1(): unsupported interpolation type" );
316
317 const quasi_unwrap<T1> X_tmp( X.get_ref());
318 const quasi_unwrap<T2> Y_tmp( Y.get_ref());
319 const quasi_unwrap<T3> XI_tmp(XI.get_ref());
320
321 if( X_tmp.is_alias(YI) || Y_tmp.is_alias(YI) || XI_tmp.is_alias(YI) )
322 {
323 Mat<eT> tmp;
324
325 interp1_helper(X_tmp.M, Y_tmp.M, XI_tmp.M, tmp, sig, extrap_val);
326
327 YI.steal_mem(tmp);
328 }
329 else
330 {
331 interp1_helper(X_tmp.M, Y_tmp.M, XI_tmp.M, YI, sig, extrap_val);
332 }
333 }

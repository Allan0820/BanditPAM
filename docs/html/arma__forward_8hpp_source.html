271struct state_type
272 {
273 #if defined(ARMA_USE_OPENMP)
274 int state;
275 #elif (!defined(ARMA_DONT_USE_STD_MUTEX))
276 std::atomic<int> state;
277 #else
278 int state;
279 #endif
280
281 arma_inline state_type() : state(int(0)) {}
282
283 // openmp: "omp atomic" does an implicit flush on the affected variable
284 // C++11: std::atomic<>::load() and std::atomic<>::store() use std::memory_order_seq_cst by default, which has an implied fence
285
286 arma_inline
287 operator int () const
288 {
289 int out;
290
291 #if defined(ARMA_USE_OPENMP)
292 #pragma omp atomic read
293 out = state;
294 #elif (!defined(ARMA_DONT_USE_STD_MUTEX))
295 out = state.load();
296 #else
297 out = state;
298 #endif
299
300 return out;
301 }
302
303 arma_inline
304 void
305 operator= (const int in_state)
306 {
307 #if defined(ARMA_USE_OPENMP)
308 #pragma omp atomic write
309 state = in_state;
310 #elif (!defined(ARMA_DONT_USE_STD_MUTEX))
311 state.store(in_state);
312 #else
313 state = in_state;
314 #endif
315 }
316 };

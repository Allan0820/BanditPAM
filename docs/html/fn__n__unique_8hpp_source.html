46template<typename T1, typename T2, typename op_n_unique_type>
47arma_hot
48inline
49uword
50n_unique
51 (
52 const SpProxy<T1>& pa,
53 const SpProxy<T2>& pb,
54 const op_n_unique_type junk
55 )
56 {
57 arma_extra_debug_sigprint();
58 arma_ignore(junk);
59
60 typename SpProxy<T1>::const_iterator_type x_it = pa.begin();
61 typename SpProxy<T1>::const_iterator_type x_it_end = pa.end();
62
63 typename SpProxy<T2>::const_iterator_type y_it = pb.begin();
64 typename SpProxy<T2>::const_iterator_type y_it_end = pb.end();
65
66 uword total_n_nonzero = 0;
67
68 while( (x_it != x_it_end) || (y_it != y_it_end) )
69 {
70 if(x_it == y_it)
71 {
72 if(op_n_unique_type::eval((*x_it), (*y_it)) != typename T1::elem_type(0))
73 {
74 ++total_n_nonzero;
75 }
76
77 ++x_it;
78 ++y_it;
79 }
80 else
81 {
82 if((x_it.col() < y_it.col()) || ((x_it.col() == y_it.col()) && (x_it.row() < y_it.row()))) // if y is closer to the end
83 {
84 if(op_n_unique_type::eval((*x_it), typename T1::elem_type(0)) != typename T1::elem_type(0))
85 {
86 ++total_n_nonzero;
87 }
88
89 ++x_it;
90 }
91 else // x is closer to the end
92 {
93 if(op_n_unique_type::eval(typename T1::elem_type(0), (*y_it)) != typename T1::elem_type(0))
94 {
95 ++total_n_nonzero;
96 }
97
98 ++y_it;
99 }
100 }
101 }
102
103 return total_n_nonzero;
104 }

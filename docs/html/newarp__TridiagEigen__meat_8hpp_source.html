45template<typename eT>
46inline
47void
48TridiagEigen<eT>::compute(const Mat<eT>& mat_obj)
49 {
50 arma_extra_debug_sigprint();
51
52 arma_debug_check( (mat_obj.is_square() == false), "newarp::TridiagEigen::compute(): matrix must be square" );
53
54 n = blas_int(mat_obj.n_rows);
55
56 main_diag = mat_obj.diag();
57 sub_diag = mat_obj.diag(-1);
58
59 evecs.set_size(n, n);
60
61 char compz = 'I';
62 blas_int lwork = blas_int(-1);
63 eT lwork_opt = eT(0);
64
65 blas_int liwork = blas_int(-1);
66 blas_int liwork_opt = blas_int(0);
67 blas_int info = blas_int(0);
68
69 // query for lwork and liwork
70 lapack::stedc(&compz, &n, main_diag.memptr(), sub_diag.memptr(), evecs.memptr(), &n, &lwork_opt, &lwork, &liwork_opt, &liwork, &info);
71
72 if(info == 0)
73 {
74 lwork = blas_int(lwork_opt);
75 liwork = liwork_opt;
76 }
77 else
78 {
79 lwork = 1 + 4 * n + n * n;
80 liwork = 3 + 5 * n;
81 }
82
83 info = blas_int(0);
84
85 podarray<eT> work(static_cast<uword>(lwork) );
86 podarray<blas_int> iwork(static_cast<uword>(liwork));
87
88 lapack::stedc(&compz, &n, main_diag.memptr(), sub_diag.memptr(), evecs.memptr(), &n, work.memptr(), &lwork, iwork.memptr(), &liwork, &info);
89
90 if(info < 0) { arma_stop_logic_error("lapack::stedc(): illegal value"); return; }
91
92 if(info > 0) { arma_stop_runtime_error("lapack::stedc(): failed to compute all eigenvalues"); return; }
93
94 computed = true;
95 }

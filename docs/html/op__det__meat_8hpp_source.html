22template<typename T1>
23inline
24bool
25op_det::apply_direct(typename T1::elem_type& out_val, const Base<typename T1::elem_type,T1>& expr)
26 {
27 arma_extra_debug_sigprint();
28
29 typedef typename T1::elem_type eT;
30 typedef typename T1::pod_type T;
31
32 if(strip_diagmat<T1>::do_diagmat)
33 {
34 const strip_diagmat<T1> strip(expr.get_ref());
35
36 out_val = op_det::apply_diagmat(strip.M);
37
38 return true;
39 }
40
41 if(strip_trimat<T1>::do_trimat)
42 {
43 const strip_trimat<T1> strip(expr.get_ref());
44
45 out_val = op_det::apply_trimat(strip.M);
46
47 return true;
48 }
49
50 Mat<eT> A(expr.get_ref());
51
52 arma_debug_check( (A.is_square() == false), "det(): given matrix must be square sized" );
53
54 if((A.n_rows <= 4) && is_cx<eT>::no)
55 {
56 constexpr T det_min = std::numeric_limits<T>::epsilon();
57 constexpr T det_max = T(1) / std::numeric_limits<T>::epsilon();
58
59 const eT det_val = op_det::apply_tiny(A);
60 const T abs_det_val = std::abs(det_val);
61
62 if((abs_det_val > det_min) && (abs_det_val < det_max)) { out_val = det_val; return true; }
63
64 // fallthrough if det_val is suspect
65 }
66
67 if(A.is_diagmat()) { out_val = op_det::apply_diagmat(A); return true; }
68
69 const bool is_triu = trimat_helper::is_triu(A);
70 const bool is_tril = is_triu ? false : trimat_helper::is_tril(A);
71
72 if(is_triu || is_tril) { out_val = op_det::apply_trimat(A); return true; }
73
74 return auxlib::det(out_val, A);
75 }

79template<typename T1>
80inline
81typename T1::elem_type
82op_det::apply_diagmat(const Base<typename T1::elem_type,T1>& expr)
83 {
84 arma_extra_debug_sigprint();
85
86 typedef typename T1::elem_type eT;
87
88 const diagmat_proxy<T1> A(expr.get_ref());
89
90 arma_debug_check( (A.n_rows != A.n_cols), "det(): given matrix must be square sized" );
91
92 const uword N = (std::min)(A.n_rows, A.n_cols);
93
94 eT val = eT(1);
95
96 for(uword i=0; i<N; ++i) { val *= A[i]; }
97
98 return val;
99 }

108 arma_extra_debug_sigprint();

111
112 const Proxy<T1> P(expr.get_ref());
113
114 const uword N = P.get_n_rows();
115
116 arma_debug_check( (N != P.get_n_cols()), "det(): given matrix must be square sized" );
117

119
120 for(uword i=0; i<N; ++i) { val *= P.at(i,i); }

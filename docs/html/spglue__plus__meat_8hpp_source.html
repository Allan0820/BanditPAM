53template<typename eT, typename T1, typename T2>
54arma_hot
55inline
56void
57spglue_plus::apply_noalias(SpMat<eT>& out, const SpProxy<T1>& pa, const SpProxy<T2>& pb)
58 {
59 arma_extra_debug_sigprint();
60
61 arma_debug_assert_same_size(pa.get_n_rows(), pa.get_n_cols(), pb.get_n_rows(), pb.get_n_cols(), "addition");
62
63 if(pa.get_n_nonzero() == 0) { out = pb.Q; return; }
64 if(pb.get_n_nonzero() == 0) { out = pa.Q; return; }
65
66 const uword max_n_nonzero = pa.get_n_nonzero() + pb.get_n_nonzero();
67
68 // Resize memory to upper bound
69 out.reserve(pa.get_n_rows(), pa.get_n_cols(), max_n_nonzero);
70
71 // Now iterate across both matrices.
72 typename SpProxy<T1>::const_iterator_type x_it = pa.begin();
73 typename SpProxy<T1>::const_iterator_type x_end = pa.end();
74
75 typename SpProxy<T2>::const_iterator_type y_it = pb.begin();
76 typename SpProxy<T2>::const_iterator_type y_end = pb.end();
77
78 uword count = 0;
79
80 while( (x_it != x_end) || (y_it != y_end) )
81 {
82 eT out_val;
83
84 const uword x_it_col = x_it.col();
85 const uword x_it_row = x_it.row();
86
87 const uword y_it_col = y_it.col();
88 const uword y_it_row = y_it.row();
89
90 bool use_y_loc = false;
91
92 if(x_it == y_it)
93 {
94 out_val = (*x_it) + (*y_it);
95
96 ++x_it;
97 ++y_it;
98 }
99 else
100 {
101 if((x_it_col < y_it_col) || ((x_it_col == y_it_col) && (x_it_row < y_it_row))) // if y is closer to the end
102 {
103 out_val = (*x_it);
104
105 ++x_it;
106 }
107 else
108 {
109 out_val = (*y_it);
110
111 ++y_it;
112
113 use_y_loc = true;
114 }
115 }
116
117 if(out_val != eT(0))
118 {
119 access::rw(out.values[count]) = out_val;
120
121 const uword out_row = (use_y_loc == false) ? x_it_row : y_it_row;
122 const uword out_col = (use_y_loc == false) ? x_it_col : y_it_col;
123
124 access::rw(out.row_indices[count]) = out_row;
125 access::rw(out.col_ptrs[out_col + 1])++;
126 ++count;
127 }
128
129 arma_check( (count > max_n_nonzero), "internal error: spglue_plus::apply_noalias(): count > max_n_nonzero" );
130 }
131
132 const uword out_n_cols = out.n_cols;
133
134 uword* col_ptrs = access::rwp(out.col_ptrs);
135
136 // Fix column pointers to be cumulative.
137 for(uword c = 1; c <= out_n_cols; ++c)
138 {
139 col_ptrs[c] += col_ptrs[c - 1];
140 }
141
142 if(count < max_n_nonzero)
143 {
144 if(count <= (max_n_nonzero/2))
145 {
146 out.mem_resize(count);
147 }
148 else
149 {
150 // quick resize without reallocating memory and copying data
151 access::rw( out.n_nonzero) = count;
152 access::rw( out.values[count]) = eT(0);
153 access::rw(out.row_indices[count]) = uword(0);
154 }
155 }
156 }

180template<typename T1, typename T2>
181inline
182void
183spglue_plus_mixed::apply(SpMat<typename eT_promoter<T1,T2>::eT>& out, const mtSpGlue<typename eT_promoter<T1,T2>::eT, T1, T2, spglue_plus_mixed>& expr)
184 {
185 arma_extra_debug_sigprint();
186
187 typedef typename T1::elem_type eT1;
188 typedef typename T2::elem_type eT2;
189
190 typedef typename promote_type<eT1,eT2>::result out_eT;
191
192 promote_type<eT1,eT2>::check();
193
194 if( (is_same_type<eT1,out_eT>::no) && (is_same_type<eT2,out_eT>::yes) )
195 {
196 // upgrade T1
197
198 const unwrap_spmat<T1> UA(expr.A);
199 const unwrap_spmat<T2> UB(expr.B);
200
201 const SpMat<eT1>& A = UA.M;
202 const SpMat<eT2>& B = UB.M;
203
204 SpMat<out_eT> AA(arma_layout_indicator(), A);
205
206 for(uword i=0; i < A.n_nonzero; ++i) { access::rw(AA.values[i]) = out_eT(A.values[i]); }
207
208 const SpMat<out_eT>& BB = reinterpret_cast< const SpMat<out_eT>& >(B);
209
210 out = AA + BB;
211 }
212 else
213 if( (is_same_type<eT1,out_eT>::yes) && (is_same_type<eT2,out_eT>::no) )
214 {
215 // upgrade T2
216
217 const unwrap_spmat<T1> UA(expr.A);
218 const unwrap_spmat<T2> UB(expr.B);
219
220 const SpMat<eT1>& A = UA.M;
221 const SpMat<eT2>& B = UB.M;
222
223 const SpMat<out_eT>& AA = reinterpret_cast< const SpMat<out_eT>& >(A);
224
225 SpMat<out_eT> BB(arma_layout_indicator(), B);
226
227 for(uword i=0; i < B.n_nonzero; ++i) { access::rw(BB.values[i]) = out_eT(B.values[i]); }
228
229 out = AA + BB;
230 }
231 else
232 {
233 // upgrade T1 and T2
234
235 const unwrap_spmat<T1> UA(expr.A);
236 const unwrap_spmat<T2> UB(expr.B);
237
238 const SpMat<eT1>& A = UA.M;
239 const SpMat<eT2>& B = UB.M;
240
241 SpMat<out_eT> AA(arma_layout_indicator(), A);
242 SpMat<out_eT> BB(arma_layout_indicator(), B);
243
244 for(uword i=0; i < A.n_nonzero; ++i) { access::rw(AA.values[i]) = out_eT(A.values[i]); }
245 for(uword i=0; i < B.n_nonzero; ++i) { access::rw(BB.values[i]) = out_eT(B.values[i]); }
246
247 out = AA + BB;
248 }
249 }

255void
256spglue_plus_mixed::dense_plus_sparse(Mat< typename promote_type<typename T1::elem_type, typename T2::elem_type >::result>& out, const T1& X, const T2& Y)
257 {
258 arma_extra_debug_sigprint();

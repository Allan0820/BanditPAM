30 typedef typename T1::elem_type eT;
31
32 const uword out_n_rows = in.aux_uword_a;
33 const uword out_n_cols = in.aux_uword_b;
34
35 const unwrap<T1> tmp(in.m);
36 const Mat<eT>& A = tmp.M;
37
38 const uword A_n_rows = A.n_rows;
39 const uword A_n_cols = A.n_cols;
40
41 const bool alias = (&actual_out == &A);
42
43 if(alias)
44 {
45 if( (A_n_rows == out_n_rows) && (A_n_cols == out_n_cols) )
46 {
47 return;
48 }
49
50 if(actual_out.is_empty())
51 {
52 actual_out.zeros(out_n_rows, out_n_cols);
53 return;
54 }
55 }
56
57 Mat<eT> B;
58 Mat<eT>& out = alias ? B : actual_out;
59
60 out.set_size(out_n_rows, out_n_cols);
61
62 if( (out_n_rows > A_n_rows) || (out_n_cols > A_n_cols) )
63 {
64 out.zeros();
65 }
66

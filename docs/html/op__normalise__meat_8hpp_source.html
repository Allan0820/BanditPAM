25void
26op_normalise_vec::apply(Mat<typename T1::elem_type>& out, const Op<T1,op_normalise_vec>& in)
27 {
28 arma_extra_debug_sigprint();
29
30 typedef typename T1::elem_type eT;
31 typedef typename T1::pod_type T;
32
33 const uword p = in.aux_uword_a;
34
35 arma_debug_check( (p == 0), "normalise(): parameter 'p' must be greater than zero" );
36
37 const quasi_unwrap<T1> U(in.m);
38
39 const T norm_val_a = norm(U.M, p);
40 const T norm_val_b = (norm_val_a != T(0)) ? norm_val_a : T(1);
41
42 if(quasi_unwrap<T1>::has_subview && U.is_alias(out))
43 {
44 Mat<eT> tmp = U.M / norm_val_b;
45
46 out.steal_mem(tmp);
47 }
48 else
49 {
50 out = U.M / norm_val_b;
51 }
52 }

89template<typename eT>
90inline
91void
92op_normalise_mat::apply(Mat<eT>& out, const Mat<eT>& A, const uword p, const uword dim)
93 {
94 arma_extra_debug_sigprint();
95
96 typedef typename get_pod_type<eT>::result T;
97
98 out.copy_size(A);
99
100 if(A.n_elem == 0) { return; }
101
102 if(dim == 0)
103 {
104 const uword n_cols = A.n_cols;
105
106 for(uword i=0; i<n_cols; ++i)
107 {
108 const T norm_val_a = norm(A.col(i), p);
109 const T norm_val_b = (norm_val_a != T(0)) ? norm_val_a : T(1);
110
111 out.col(i) = A.col(i) / norm_val_b;
112 }
113 }
114 else
115 {
116 const uword n_rows = A.n_rows;
117 const uword n_cols = A.n_cols;
118
119 podarray<T> norm_vals(n_rows);
120
121 T* norm_vals_mem = norm_vals.memptr();
122
123 for(uword i=0; i<n_rows; ++i)
124 {
125 const T norm_val = norm(A.row(i), p);
126
127 norm_vals_mem[i] = (norm_val != T(0)) ? norm_val : T(1);
128 }
129
130 const eT* A_mem = A.memptr();
131 eT* out_mem = out.memptr();
132
133 for(uword col=0; col < n_cols; ++col)
134 for(uword row=0; row < n_rows; ++row)
135 {
136 (*out_mem) = (*A_mem) / norm_vals_mem[row];
137
138 A_mem++;
139 out_mem++;
140 }
141 }
142 }

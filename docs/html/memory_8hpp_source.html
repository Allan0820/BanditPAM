36template<typename eT>
37inline
38arma_malloc
39eT*
40memory::acquire(const uword n_elem)
41 {
42 if(n_elem == 0) { return nullptr; }
43
44 arma_debug_check
45 (
46 ( size_t(n_elem) > (std::numeric_limits<size_t>::max() / sizeof(eT)) ),
47 "arma::memory::acquire(): requested size is too large"
48 );
49
50 eT* out_memptr;
51
52 #if defined(ARMA_USE_TBB_ALLOC)
53 {
54 out_memptr = (eT *) scalable_malloc(sizeof(eT)*n_elem);
55 }
56 #elif defined(ARMA_USE_MKL_ALLOC)
57 {
58 out_memptr = (eT *) mkl_malloc( sizeof(eT)*n_elem, 32 );
59 }
60 #elif defined(ARMA_USE_NUMPY_ALLOC)
61 {
62 out_memptr = (eT *) npy_malloc(sizeof(eT) * n_elem);
63 }
64 #elif defined(ARMA_HAVE_POSIX_MEMALIGN)
65 {
66 eT* memptr = nullptr;
67
68 const size_t n_bytes = sizeof(eT)*size_t(n_elem);
69 const size_t alignment = (n_bytes >= size_t(1024)) ? size_t(32) : size_t(16);
70
71 // TODO: investigate apparent memory leak when using alignment >= 64 (as shown on Fedora 28, glibc 2.27)
72 int status = posix_memalign((void **)&memptr, ( (alignment >= sizeof(void*)) ? alignment : sizeof(void*) ), n_bytes);
73
74 out_memptr = (status == 0) ? memptr : nullptr;
75 }
76 #elif defined(_MSC_VER)
77 {
78 //out_memptr = (eT *) malloc(sizeof(eT)*n_elem);
79 //out_memptr = (eT *) _aligned_malloc( sizeof(eT)*n_elem, 16 ); // lives in malloc.h
80
81 const size_t n_bytes = sizeof(eT)*size_t(n_elem);
82 const size_t alignment = (n_bytes >= size_t(1024)) ? size_t(32) : size_t(16);
83
84 out_memptr = (eT *) _aligned_malloc( n_bytes, alignment );
85 }
86 #else
87 {
88 //return ( new(std::nothrow) eT[n_elem] );
89 out_memptr = (eT *) malloc(sizeof(eT)*n_elem);
90 }
91 #endif
92
93 // TODO: for mingw, use __mingw_aligned_malloc
94
95 arma_check_bad_alloc( (out_memptr == nullptr), "arma::memory::acquire(): out of memory" );
96
97 return out_memptr;
98 }

146 {
147 #if (defined(ARMA_HAVE_ICC_ASSUME_ALIGNED) || defined(ARMA_HAVE_GCC_ASSUME_ALIGNED)) && !defined(ARMA_DONT_CHECK_ALIGNMENT)

163arma_inline
164void
165memory::mark_as_aligned(eT*& mem)
166 {
167 #if defined(ARMA_HAVE_ICC_ASSUME_ALIGNED)
168 {
169 __assume_aligned(mem, 16);
170 }
171 #elif defined(ARMA_HAVE_GCC_ASSUME_ALIGNED)
172 {
173 mem = (eT*)__builtin_assume_aligned(mem, 16);
174 }
175 #else
176 {
177 arma_ignore(mem);
178 }
179 #endif
180
181 // TODO: MSVC? __assume( (mem & 0x0F) == 0 );
182 //
183 // http://comments.gmane.org/gmane.comp.gcc.patches/239430
184 // GCC __builtin_assume_aligned is similar to ICC's __assume_aligned,
185 // so for lvalue first argument ICC's __assume_aligned can be emulated using
186 // #define __assume_aligned(lvalueptr, align) lvalueptr = __builtin_assume_aligned (lvalueptr, align)
187 //
188 // http://www.inf.ethz.ch/personal/markusp/teaching/263-2300-ETH-spring11/slides/class19.pdf
189 // http://software.intel.com/sites/products/documentation/hpc/composerxe/en-us/cpp/lin/index.htm
190 // http://d3f8ykwhia686p.cloudfront.net/1live/intel/CompilerAutovectorizationGuide.pdf

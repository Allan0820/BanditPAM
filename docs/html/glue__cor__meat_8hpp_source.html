25glue_cor::apply(Mat<typename T1::elem_type>& out, const Glue<T1,T2,glue_cor>& X)
26 {
27 arma_extra_debug_sigprint();
28
29 typedef typename T1::elem_type eT;
30
31 const uword norm_type = X.aux_uword;
32
33 const unwrap<T1> UA(X.A);
34 const unwrap<T2> UB(X.B);
35
36 const Mat<eT>& A = UA.M;
37 const Mat<eT>& B = UB.M;
38
39 const Mat<eT>& AA = (A.n_rows == 1)
40 ? Mat<eT>(const_cast<eT*>(A.memptr()), A.n_cols, A.n_rows, false, false)
41 : Mat<eT>(const_cast<eT*>(A.memptr()), A.n_rows, A.n_cols, false, false);
42
43 const Mat<eT>& BB = (B.n_rows == 1)
44 ? Mat<eT>(const_cast<eT*>(B.memptr()), B.n_cols, B.n_rows, false, false)
45 : Mat<eT>(const_cast<eT*>(B.memptr()), B.n_rows, B.n_cols, false, false);
46
47 arma_debug_assert_mul_size(AA, BB, true, false, "cor()");
48
49 if( (AA.n_elem == 0) || (BB.n_elem == 0) )
50 {
51 out.reset();
52 return;
53 }
54
55 const uword N = AA.n_rows;
56 const eT norm_val = (norm_type == 0) ? ( (N > 1) ? eT(N-1) : eT(1) ) : eT(N);
57
58 const Mat<eT> tmp1 = AA.each_row() - mean(AA,0);
59 const Mat<eT> tmp2 = BB.each_row() - mean(BB,0);
60
61 out = tmp1.t() * tmp2;
62 out /= norm_val;
63
64 out /= conv_to< Mat<eT> >::from( stddev(AA).t() * stddev(BB) ); // TODO: check for zeros?
65 }

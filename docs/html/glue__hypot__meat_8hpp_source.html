25glue_hypot::apply(Mat<typename T1::elem_type>& out, const Glue<T1, T2, glue_hypot>& expr)
26 {
27 arma_extra_debug_sigprint();
28
29 typedef typename T1::elem_type eT;
30
31 const Proxy<T1> P1(expr.A);
32 const Proxy<T2> P2(expr.B);
33
34 arma_assert_same_size(P1, P2, "hypot()");
35
36 const bool bad_alias = ( (Proxy<T1>::has_subview && P1.is_alias(out)) || (Proxy<T2>::has_subview && P2.is_alias(out)) );
37

39 {
40 glue_hypot::apply_noalias(out, P1, P2);
41 }
42 else

44 Mat<eT> tmp;

46 glue_hypot::apply_noalias(tmp, P1, P2);

55inline
56void
57glue_hypot::apply_noalias(Mat<typename T1::elem_type>& out, const Proxy<T1>& P1, const Proxy<T2>& P2)
58 {
59 arma_extra_debug_sigprint();
60
61 typedef typename T1::elem_type eT;
62
63 const uword n_rows = P1.get_n_rows();
64 const uword n_cols = P1.get_n_cols();
65
66 out.set_size(n_rows, n_cols);
67
68 eT* out_mem = out.memptr();
69
70 if( (Proxy<T1>::use_at == false) && (Proxy<T2>::use_at == false) )
71 {
72 typename Proxy<T1>::ea_type eaP1 = P1.get_ea();
73 typename Proxy<T2>::ea_type eaP2 = P2.get_ea();
74
75 const uword N = P1.get_n_elem();
76
77 for(uword i=0; i<N; ++i)
78 {
79 out_mem[i] = arma_hypot( eaP1[i], eaP2[i] );
80 }
81 }
82 else
83 {
84 for(uword col=0; col < n_cols; ++col)
85 for(uword row=0; row < n_rows; ++row)
86 {
87 *out_mem = arma_hypot( P1.at(row,col), P2.at(row,col) );
88 out_mem++;
89 }
90 }
91 }

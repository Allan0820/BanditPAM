22template<typename eT>
23inline
24void
25inplace_strans
26 (
27 Mat<eT>& X,
28 const char* method = "std"
29 )
30 {
31 arma_extra_debug_sigprint();
32
33 const char sig = (method != nullptr) ? method[0] : char(0);
34
35 arma_debug_check( ((sig != 's') && (sig != 'l')), "inplace_strans(): unknown method specified" );
36
37 const bool low_memory = (sig == 'l');
38
39 if( (low_memory == false) || (X.n_rows == X.n_cols) )
40 {
41 op_strans::apply_mat_inplace(X);
42 }
43 else
44 {
45 // in-place algorithm inspired by:
46 // Fred G. Gustavson, Tadeusz Swirszcz.
47 // In-Place Transposition of Rectangular Matrices.
48 // Applied Parallel Computing. State of the Art in Scientific Computing.
49 // Lecture Notes in Computer Science. Volume 4699, pp. 560-569, 2007.
50
51
52 // X.set_size() will check whether we can change the dimensions of X;
53 // X.set_size() will also reuse existing memory, as the number of elements hasn't changed
54
55 X.set_size(X.n_cols, X.n_rows);
56
57 const uword m = X.n_cols;
58 const uword n = X.n_rows;
59
60 std::vector<bool> visited(X.n_elem); // TODO: replace std::vector<bool> with a better implementation
61
62 for(uword col = 0; col < m; ++col)
63 for(uword row = 0; row < n; ++row)
64 {
65 const uword pos = col*n + row;
66
67 if(visited[pos] == false)
68 {
69 uword curr_pos = pos;
70
71 eT val = X.at(row, col);
72
73 while(visited[curr_pos] == false)
74 {
75 visited[curr_pos] = true;
76
77 const uword j = curr_pos / m;
78 const uword i = curr_pos - m * j;
79
80 const eT tmp = X.at(j, i);
81 X.at(j, i) = val;
82 val = tmp;
83
84 curr_pos = i*n + j;
85 }
86 }
87 }
88 }
89 }

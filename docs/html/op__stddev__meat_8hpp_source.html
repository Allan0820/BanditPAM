28op_stddev::apply(Mat<typename T1::pod_type>& out, const mtOp<typename T1::pod_type, T1, op_stddev>& in)
29 {
30 arma_extra_debug_sigprint();
31
32 typedef typename T1::elem_type in_eT;
33 typedef typename T1::pod_type out_eT;
34
35 const unwrap_check_mixed<T1> tmp(in.m, out);
36 const Mat<in_eT>& X = tmp.M;
37
38 const uword norm_type = in.aux_uword_a;
39 const uword dim = in.aux_uword_b;
40
41 arma_debug_check( (norm_type > 1), "stddev(): parameter 'norm_type' must be 0 or 1" );
42 arma_debug_check( (dim > 1), "stddev(): parameter 'dim' must be 0 or 1" );
43
44 const uword X_n_rows = X.n_rows;
45 const uword X_n_cols = X.n_cols;
46
47 if(dim == 0)
48 {
49 arma_extra_debug_print("op_stddev::apply(): dim = 0");
50
51 out.set_size((X_n_rows > 0) ? 1 : 0, X_n_cols);
52
53 if(X_n_rows > 0)
54 {
55 out_eT* out_mem = out.memptr();
56
57 for(uword col=0; col<X_n_cols; ++col)
58 {
59 out_mem[col] = std::sqrt( op_var::direct_var( X.colptr(col), X_n_rows, norm_type ) );
60 }
61 }
62 }
63 else
64 if(dim == 1)
65 {
66 arma_extra_debug_print("op_stddev::apply(): dim = 1");
67
68 out.set_size(X_n_rows, (X_n_cols > 0) ? 1 : 0);
69
70 if(X_n_cols > 0)
71 {
72 podarray<in_eT> dat(X_n_cols);
73
74 in_eT* dat_mem = dat.memptr();
75 out_eT* out_mem = out.memptr();
76
77 for(uword row=0; row<X_n_rows; ++row)
78 {
79 dat.copy_row(X, row);
80
81 out_mem[row] = std::sqrt( op_var::direct_var( dat_mem, X_n_cols, norm_type) );
82 }
83 }
84 }
85 }

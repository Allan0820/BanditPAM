50template<typename T1>
51inline
52typename T1::pod_type
53op_cond::rcond(const Base<typename T1::elem_type, T1>& X)
54 {
55 arma_extra_debug_sigprint();
56
57 typedef typename T1::elem_type eT;
58 typedef typename T1::pod_type T;
59
60 if(strip_trimat<T1>::do_trimat)
61 {
62 const strip_trimat<T1> S(X.get_ref());
63
64 const quasi_unwrap<typename strip_trimat<T1>::stored_type> U(S.M);
65
66 arma_debug_check( (U.M.is_square() == false), "rcond(): matrix must be square sized" );
67
68 const uword layout = (S.do_triu) ? uword(0) : uword(1);
69
70 return auxlib::rcond_trimat(U.M, layout);
71 }
72
73 Mat<eT> A = X.get_ref();
74
75 arma_debug_check( (A.is_square() == false), "rcond(): matrix must be square sized" );
76
77 if(A.is_empty()) { return Datum<T>::inf; }
78
79 const bool is_triu = trimat_helper::is_triu(A);
80 const bool is_tril = (is_triu) ? false : trimat_helper::is_tril(A);
81
82 if(is_triu || is_tril)
83 {
84 const uword layout = (is_triu) ? uword(0) : uword(1);
85
86 return auxlib::rcond_trimat(A, layout);
87 }
88
89 #if defined(ARMA_OPTIMISE_SYMPD)
90 const bool try_sympd = auxlib::crippled_lapack(A) ? false : sympd_helper::guess_sympd_anysize(A);
91 #else
92 const bool try_sympd = false;
93 #endif
94
95 if(try_sympd)
96 {
97 bool calc_ok = false;
98
99 const T out_val = auxlib::rcond_sympd(A, calc_ok);
100
101 if(calc_ok) { return out_val; }
102
103 // auxlib::rcond_sympd() may have failed because A isn't really sympd
104 // restore A, as auxlib::rcond_sympd() may have destroyed it
105 A = X.get_ref();
106 // fallthrough to the next return statement
107 }
108
109 return auxlib::rcond(A);
110 }

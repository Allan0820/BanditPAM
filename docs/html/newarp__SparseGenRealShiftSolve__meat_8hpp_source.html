21template<typename eT>
22inline
23SparseGenRealShiftSolve<eT>::SparseGenRealShiftSolve(const SpMat<eT>& mat_obj, const eT shift)
24 #if defined(ARMA_USE_SUPERLU)
25 : perm_c(mat_obj.n_cols + 1)
26 , perm_r(mat_obj.n_rows + 1)
27 , n_rows(mat_obj.n_rows)
28 , n_cols(mat_obj.n_cols)
29 #endif
30 {
31 arma_extra_debug_sigprint();
32
33 #if defined(ARMA_USE_SUPERLU)
34 {
35 // Derived from sp_auxlib::run_aupd_shiftinvert()
36 superlu_opts superlu_opts_default;
37 superlu::superlu_options_t options;
38 sp_auxlib::set_superlu_opts(options, superlu_opts_default);
39
40 superlu::GlobalLU_t Glu;
41 arrayops::fill_zeros(reinterpret_cast<char*>(&Glu), sizeof(superlu::GlobalLU_t));
42
43 superlu_supermatrix_wrangler x;
44 superlu_supermatrix_wrangler xC;
45 superlu_array_wrangler<int> etree(mat_obj.n_cols+1);
46
47 // Copy A-shift*I to x
48 const bool status_x = sp_auxlib::copy_to_supermatrix_with_shift(x.get_ref(), mat_obj, shift);
49
50 if(status_x == false) { arma_stop_runtime_error("newarp::SparseGenRealShiftSolve::SparseGenRealShiftSolve(): could not construct SuperLU matrix"); return; }
51
52 int panel_size = superlu::sp_ispec_environ(1);
53 int relax = superlu::sp_ispec_environ(2);
54 int slu_info = 0; // Return code
55 int lwork = 0; // lwork = 0: allocate space internally by system malloc
56
57 superlu_stat_wrangler stat;
58
59 arma_extra_debug_print("superlu::gstrf()");
60 superlu::get_permutation_c(options.ColPerm, x.get_ptr(), perm_c.get_ptr());
61 superlu::sp_preorder_mat(&options, x.get_ptr(), perm_c.get_ptr(), etree.get_ptr(), xC.get_ptr());
62 superlu::gstrf<eT>(&options, xC.get_ptr(), relax, panel_size, etree.get_ptr(), NULL, lwork, perm_c.get_ptr(), perm_r.get_ptr(), l.get_ptr(), u.get_ptr(), &Glu, stat.get_ptr(), &slu_info);
63
64 if(slu_info != 0)
65 {
66 arma_debug_warn_level(2, "matrix is singular to working precision");
67 return;
68 }
69
70 eT x_norm_val = sp_auxlib::norm1<eT>(x.get_ptr());
71 eT x_rcond = sp_auxlib::lu_rcond<eT>(l.get_ptr(), u.get_ptr(), x_norm_val);
72
73 if( (x_rcond < std::numeric_limits<eT>::epsilon()) || arma_isnan(x_rcond) )
74 {
75 if(x_rcond > eT(0)) { arma_debug_warn_level(2, "matrix is singular to working precision (rcond: ", x_rcond, ")"); }
76 else { arma_debug_warn_level(2, "matrix is singular to working precision"); }
77 return;
78 }
79
80 valid = true;
81 }
82 #else
83 {
84 arma_ignore(mat_obj);
85 arma_ignore(shift);
86 }
87 #endif
88 }

92// Perform the shift-solve operation \f$y=(A-\sigma I)^{-1}x\f$.
93// y_out = inv(A - sigma * I) * x_in
94template<typename eT>
95inline
96void
97SparseGenRealShiftSolve<eT>::perform_op(eT* x_in, eT* y_out) const
98 {
99 arma_extra_debug_sigprint();
100
101 #if defined(ARMA_USE_SUPERLU)
102 {
103 const Col<eT> x(x_in , n_cols, false, true);
104 Col<eT> y(y_out, n_rows, false, true);
105
106 // Derived from sp_auxlib::run_aupd_shiftinvert()
107 y = x;
108 superlu_supermatrix_wrangler out_slu;
109
110 const bool status_out_slu = sp_auxlib::wrap_to_supermatrix(out_slu.get_ref(), y);
111
112 if(status_out_slu == false) { arma_stop_runtime_error("newarp::SparseGenRealShiftSolve::perform_op(): could not construct SuperLU matrix"); return; }
113
114 superlu_stat_wrangler stat;
115 int info = 0;
116
117 arma_extra_debug_print("superlu::gstrs()");
118 superlu::gstrs<eT>(superlu::NOTRANS, l.get_ptr(), u.get_ptr(), perm_c.get_ptr(), perm_r.get_ptr(), out_slu.get_ptr(), stat.get_ptr(), &info);
119
120 if(info != 0) { arma_stop_runtime_error("newarp::SparseGenRealShiftSolve::perform_op(): could not solve linear equation"); return; }
121
122 // No need to modify memory further since it was all done in-place.
123 }
124 #else
125 {
126 arma_ignore(x_in);
127 arma_ignore(y_out);
128 }
129 #endif
130 }

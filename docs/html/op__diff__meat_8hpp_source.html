21template<typename eT>
22inline
23void
24op_diff::apply_noalias(Mat<eT>& out, const Mat<eT>& X, const uword k, const uword dim)
25 {
26 arma_extra_debug_sigprint();
27
28 uword n_rows = X.n_rows;
29 uword n_cols = X.n_cols;
30
31 if(dim == 0)
32 {
33 if(n_rows <= k) { out.set_size(0,n_cols); return; }
34
35 --n_rows;
36
37 out.set_size(n_rows,n_cols);
38
39 for(uword col=0; col < n_cols; ++col)
40 {
41 eT* out_colmem = out.colptr(col);
42 const eT* X_colmem = X.colptr(col);
43
44 for(uword row=0; row < n_rows; ++row)
45 {
46 const eT val0 = X_colmem[row ];
47 const eT val1 = X_colmem[row+1];
48
49 out_colmem[row] = val1 - val0;
50 }
51 }
52
53 if(k >= 2)
54 {
55 for(uword iter=2; iter <= k; ++iter)
56 {
57 --n_rows;
58
59 for(uword col=0; col < n_cols; ++col)
60 {
61 eT* colmem = out.colptr(col);
62
63 for(uword row=0; row < n_rows; ++row)
64 {
65 const eT val0 = colmem[row ];
66 const eT val1 = colmem[row+1];
67
68 colmem[row] = val1 - val0;
69 }
70 }
71 }
72
73 out = out( span(0,n_rows-1), span::all );
74 }
75 }
76 else
77 if(dim == 1)
78 {
79 if(n_cols <= k) { out.set_size(n_rows,0); return; }
80
81 --n_cols;
82
83 out.set_size(n_rows,n_cols);
84
85 if(n_rows == 1)
86 {
87 const eT* X_mem = X.memptr();
88 eT* out_mem = out.memptr();
89
90 for(uword col=0; col < n_cols; ++col)
91 {
92 const eT val0 = X_mem[col ];
93 const eT val1 = X_mem[col+1];
94
95 out_mem[col] = val1 - val0;
96 }
97 }
98 else
99 {
100 for(uword col=0; col < n_cols; ++col)
101 {
102 eT* out_col_mem = out.colptr(col);
103
104 const eT* X_col0_mem = X.colptr(col );
105 const eT* X_col1_mem = X.colptr(col+1);
106
107 for(uword row=0; row < n_rows; ++row)
108 {
109 out_col_mem[row] = X_col1_mem[row] - X_col0_mem[row];
110 }
111 }
112 }
113
114 if(k >= 2)
115 {
116 for(uword iter=2; iter <= k; ++iter)
117 {
118 --n_cols;
119
120 if(n_rows == 1)
121 {
122 eT* out_mem = out.memptr();
123
124 for(uword col=0; col < n_cols; ++col)
125 {
126 const eT val0 = out_mem[col ];
127 const eT val1 = out_mem[col+1];
128
129 out_mem[col] = val1 - val0;
130 }
131 }
132 else
133 {
134 for(uword col=0; col < n_cols; ++col)
135 {
136 eT* col0_mem = out.colptr(col );
137 const eT* col1_mem = out.colptr(col+1);
138
139 for(uword row=0; row < n_rows; ++row)
140 {
141 col0_mem[row] = col1_mem[row] - col0_mem[row];
142 }
143 }
144 }
145 }
146
147 out = out( span::all, span(0,n_cols-1) );
148 }
149 }
150 }

154template<typename T1>
155inline
156void
157op_diff::apply(Mat<typename T1::elem_type>& out, const Op<T1,op_diff>& in)
158 {
159 arma_extra_debug_sigprint();
160
161 typedef typename T1::elem_type eT;
162
163 const uword k = in.aux_uword_a;
164 const uword dim = in.aux_uword_b;
165
166 arma_debug_check( (dim > 1), "diff(): parameter 'dim' must be 0 or 1" );
167
168 if(k == 0) { out = in.m; return; }
169
170 const quasi_unwrap<T1> U(in.m);
171
172 if(U.is_alias(out))
173 {
174 Mat<eT> tmp;
175
176 op_diff::apply_noalias(tmp, U.M, k, dim);
177
178 out.steal_mem(tmp);
179 }
180 else
181 {
182 op_diff::apply_noalias(out, U.M, k, dim);
183 }
184 }

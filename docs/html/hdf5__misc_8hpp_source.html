180// Compare datatype against all supported types.
181inline
182bool
183is_supported_arma_hdf5_type(hid_t datatype)
184 {
185 hid_t search_type;
186
187 bool is_equal;
188
189
190 // start with most likely used types: double, complex<double>, float, complex<float>
191
192 search_type = get_hdf5_type<double>();
193 is_equal = ( arma_H5Tequal(datatype, search_type) > 0 );
194 arma_H5Tclose(search_type);
195 if(is_equal) { return true; }
196
197 search_type = get_hdf5_type< std::complex<double> >();
198 is_equal = ( arma_H5Tequal(datatype, search_type) > 0 );
199 arma_H5Tclose(search_type);
200 if(is_equal) { return true; }
201
202 search_type = get_hdf5_type<float>();
203 is_equal = ( arma_H5Tequal(datatype, search_type) > 0 );
204 arma_H5Tclose(search_type);
205 if(is_equal) { return true; }
206
207 search_type = get_hdf5_type< std::complex<float> >();
208 is_equal = ( arma_H5Tequal(datatype, search_type) > 0 );
209 arma_H5Tclose(search_type);
210 if(is_equal) { return true; }
211
212
213 // remaining supported types: u8, s8, u16, s16, u32, s32, u64, s64, ulng_t, slng_t
214
215 search_type = get_hdf5_type<u8>();
216 is_equal = ( arma_H5Tequal(datatype, search_type) > 0 );
217 arma_H5Tclose(search_type);
218 if(is_equal) { return true; }
219
220 search_type = get_hdf5_type<s8>();
221 is_equal = ( arma_H5Tequal(datatype, search_type) > 0 );
222 arma_H5Tclose(search_type);
223 if(is_equal) { return true; }
224
225 search_type = get_hdf5_type<u16>();
226 is_equal = ( arma_H5Tequal(datatype, search_type) > 0 );

271struct hdf5_search_info
272 {
273 const std::vector<std::string>& names;
274 int num_dims;
275 bool exact;
276 hid_t best_match;
277 size_t best_match_position; // Position of best match in names vector.
278 };

281
282inline
283herr_t
284hdf5_search_callback
285 (
286 hid_t loc_id,
287 const char* name,
288 const H5O_info_t* info,
289 void* operator_data // hdf5_search_info
290 )
291 {
292 hdf5_search_info* search_info = (hdf5_search_info*) operator_data;
293
294 // We are looking for datasets.
295 if(info->type == H5O_TYPE_DATASET)
296 {
297 // Check type of dataset to see if we could even load it.
298 hid_t dataset = arma_H5Dopen(loc_id, name, H5P_DEFAULT);
299 hid_t datatype = arma_H5Dget_type(dataset);
300
301 const bool is_supported = is_supported_arma_hdf5_type(datatype);
302
303 arma_H5Tclose(datatype);
304 arma_H5Dclose(dataset);
305
306 if(is_supported == false)
307 {
308 // Forget about it and move on.
309 return 0;
310 }
311
312 // Now we have to check against our set of names.
313 // Only check names which could be better.
314 for(size_t string_pos = 0; string_pos < search_info->best_match_position; ++string_pos)
315 {
316 // name is the full path (/path/to/dataset); names[string_pos] may be
317 // "dataset", "/to/dataset", or "/path/to/dataset".
318 // So if we count the number of forward slashes in names[string_pos],
319 // and then simply take the last substring of name containing that number of slashes,
320 // we can do the comparison.
321
322 // Count the number of forward slashes in names[string_pos].
323 uword name_count = 0;
324 for(uword i = 0; i < search_info->names[string_pos].length(); ++i)
325 {
326 if((search_info->names[string_pos])[i] == '/') { ++name_count; }
327 }
328
329 // Count the number of forward slashes in the full name.
330 uword count = 0;
331 const std::string str = std::string(name);
332 for(uword i = 0; i < str.length(); ++i)
333 {
334 if(str[i] == '/') { ++count; }
335 }
336
337 // Is the full string the same?
338 if(str == search_info->names[string_pos])
339 {
340 // We found it exactly.
341 hid_t match_candidate = arma_H5Dopen(loc_id, name, H5P_DEFAULT);
342
343 if(match_candidate < 0)
344 {
345 return -1;
346 }
347
348 // Ensure that the dataset is valid and of the correct dimensionality.
349 hid_t filespace = arma_H5Dget_space(match_candidate);
350 int num_dims = arma_H5Sget_simple_extent_ndims(filespace);
351
352 if(num_dims <= search_info->num_dims)
353 {
354 // Valid dataset -- we'll keep it.
355 // If we already have an existing match we have to close it.
356 if(search_info->best_match != -1)
357 {
358 arma_H5Dclose(search_info->best_match);
359 }
360
361 search_info->best_match_position = string_pos;
362 search_info->best_match = match_candidate;
363 }
364
365 arma_H5Sclose(filespace);
366 // There is no possibility of anything better, so terminate the search.
367 return 1;
368 }
369
370 // If we are asking for more slashes than we have, this can't be a match.
371 // Skip to below, where we decide whether or not to keep it anyway based
372 // on the exactness condition of the search.
373 if(count <= name_count)
374 {
375 size_t start_pos = (count == 0) ? 0 : std::string::npos;
376 while(count > 0)
377 {
378 // Move pointer to previous slash.
379 start_pos = str.rfind('/', start_pos);
380
381 // Break if we've run out of slashes.
382 if(start_pos == std::string::npos) { break; }
383
384 --count;
385 }
386
387 // Now take the substring (this may end up being the full string).
388 const std::string substring = str.substr(start_pos);
389
390 // Are they the same?
391 if(substring == search_info->names[string_pos])
392 {
393 // We have found the object; it must be better than our existing match.
394 hid_t match_candidate = arma_H5Dopen(loc_id, name, H5P_DEFAULT);
395
396
397 // arma_check(match_candidate < 0, "Mat::load(): cannot open an HDF5 dataset");
398 if(match_candidate < 0)
399 {
400 return -1;
401 }
402
403
404 // Ensure that the dataset is valid and of the correct dimensionality.
405 hid_t filespace = arma_H5Dget_space(match_candidate);
406 int num_dims = arma_H5Sget_simple_extent_ndims(filespace);
407
408 if(num_dims <= search_info->num_dims)
409 {
410 // Valid dataset -- we'll keep it.
411 // If we already have an existing match we have to close it.
412 if(search_info->best_match != -1)
413 {
414 arma_H5Dclose(search_info->best_match);
415 }
416
417 search_info->best_match_position = string_pos;
418 search_info->best_match = match_candidate;
419 }
420
421 arma_H5Sclose(filespace);
422 }
423 }
424
425
426 // If they are not the same, but we have not found anything and we don't need an exact match, take this.
427 if((search_info->exact == false) && (search_info->best_match == -1))
428 {
429 hid_t match_candidate = arma_H5Dopen(loc_id, name, H5P_DEFAULT);
430
431 // arma_check(match_candidate < 0, "Mat::load(): cannot open an HDF5 dataset");
432 if(match_candidate < 0)
433 {
434 return -1;
435 }
436
437 hid_t filespace = arma_H5Dget_space(match_candidate);
438 int num_dims = arma_H5Sget_simple_extent_ndims(filespace);
439
440 if(num_dims <= search_info->num_dims)
441 {
442 // Valid dataset -- we'll keep it.
443 search_info->best_match = arma_H5Dopen(loc_id, name, H5P_DEFAULT);
444 }
445
446 arma_H5Sclose(filespace);
447 }
448 }
449 }
450
451 return 0;
452 }

463inline
464hid_t
465search_hdf5_file
466 (
467 const std::vector<std::string>& names,
468 hid_t hdf5_file,
469 int num_dims = 2,
470 bool exact = false
471 )
472 {
473 hdf5_search_info search_info = { names, num_dims, exact, -1, names.size() };
474
475 // We'll use the H5Ovisit to track potential entries.
476 herr_t status = arma_H5Ovisit(hdf5_file, H5_INDEX_NAME, H5_ITER_NATIVE, hdf5_search_callback, void_ptr(&search_info));
477
478 // Return the best match; it will be -1 if there was a problem.
479 return (status < 0) ? -1 : search_info.best_match;
480 }

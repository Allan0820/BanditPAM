45template<typename T1>
46inline
47bool
48op_orth::apply_direct(Mat<typename T1::elem_type>& out, const Base<typename T1::elem_type,T1>& expr, typename T1::pod_type tol)
49 {
50 arma_extra_debug_sigprint();
51
52 typedef typename T1::elem_type eT;
53 typedef typename T1::pod_type T;
54
55 arma_debug_check((tol < T(0)), "orth(): tolerance must be >= 0");
56
57 Mat<eT> A(expr.get_ref());
58
59 Mat<eT> U;
60 Col< T> s;
61 Mat<eT> V;
62
63 const bool status = auxlib::svd_dc(U, s, V, A);
64
65 V.reset();
66
67 if(status == false) { return false; }
68
69 if(s.is_empty()) { out.reset(); return true; }
70
71 const uword s_n_elem = s.n_elem;
72 const T* s_mem = s.memptr();
73
74 // set tolerance to default if it hasn't been specified
75 if(tol == T(0)) { tol = (std::max)(A.n_rows, A.n_cols) * s_mem[0] * std::numeric_limits<T>::epsilon(); }
76
77 uword count = 0;
78
79 for(uword i=0; i < s_n_elem; ++i) { count += (s_mem[i] > tol) ? uword(1) : uword(0); }
80
81 if(count > 0)
82 {
83 out = U.head_cols(count); // out *= eT(-1);
84 }
85 else
86 {
87 out.set_size(A.n_rows, 0);
88 }
89
90 return true;
91 }

121template<typename T1>
122inline
123bool
124op_null::apply_direct(Mat<typename T1::elem_type>& out, const Base<typename T1::elem_type,T1>& expr, typename T1::pod_type tol)
125 {
126 arma_extra_debug_sigprint();
127
128 typedef typename T1::elem_type eT;
129 typedef typename T1::pod_type T;
130
131 arma_debug_check((tol < T(0)), "null(): tolerance must be >= 0");
132
133 Mat<eT> A(expr.get_ref());
134
135 Mat<eT> U;
136 Col< T> s;
137 Mat<eT> V;
138
139 const bool status = auxlib::svd_dc(U, s, V, A);
140
141 U.reset();
142
143 if(status == false) { return false; }
144
145 if(s.is_empty()) { out.reset(); return true; }
146
147 const uword s_n_elem = s.n_elem;
148 const T* s_mem = s.memptr();
149
150 // set tolerance to default if it hasn't been specified
151 if(tol == T(0)) { tol = (std::max)(A.n_rows, A.n_cols) * s_mem[0] * std::numeric_limits<T>::epsilon(); }
152
153 uword count = 0;
154
155 for(uword i=0; i < s_n_elem; ++i) { count += (s_mem[i] > tol) ? uword(1) : uword(0); }
156
157 if(count < A.n_cols)
158 {
159 out = V.tail_cols(A.n_cols - count);
160
161 const uword out_n_elem = out.n_elem;
162 eT* out_mem = out.memptr();
163
164 for(uword i=0; i<out_n_elem; ++i)
165 {
166 if(std::abs(out_mem[i]) < std::numeric_limits<T>::epsilon()) { out_mem[i] = eT(0); }
167 }
168 }
169 else
170 {
171 out.set_size(A.n_cols, 0);
172 }
173
174 return true;
175 }

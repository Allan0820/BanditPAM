115template<typename T1>
116inline
117typename enable_if2< is_supported_blas_type<typename T1::elem_type>::value, bool >::result
118eig_sym
119 (
120 Col<typename T1::pod_type>& eigval,
121 Mat<typename T1::elem_type>& eigvec,
122 const Base<typename T1::elem_type,T1>& expr,
123 const char* method = "dc"
124 )
125 {
126 arma_extra_debug_sigprint();
127
128 typedef typename T1::elem_type eT;
129
130 const char sig = (method != nullptr) ? method[0] : char(0);
131
132 arma_debug_check( ((sig != 's') && (sig != 'd')), "eig_sym(): unknown method specified" );
133 arma_debug_check( void_ptr(&eigval) == void_ptr(&eigvec), "eig_sym(): parameter 'eigval' is an alias of parameter 'eigvec'" );
134
135 const quasi_unwrap<T1> U(expr.get_ref());
136
137 const bool is_alias = U.is_alias(eigvec);
138
139 Mat<eT> eigvec_tmp;
140 Mat<eT>& eigvec_out = (is_alias == false) ? eigvec : eigvec_tmp;
141
142 const bool status = eig_sym_helper(eigval, eigvec_out, U.M, sig, "eig_sym()");
143
144 if(status == false)
145 {
146 eigval.soft_reset();
147 eigvec.soft_reset();
148 arma_debug_warn_level(3, "eig_sym(): decomposition failed");
149 }
150 else
151 {
152 if(is_alias) { eigvec.steal_mem(eigvec_tmp); }
153 }
154
155 return status;
156 }

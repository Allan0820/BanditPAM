71template<typename eT>
72inline
73bool
74op_roots::apply_noalias(Mat< std::complex<typename get_pod_type<eT>::result> >& out, const Mat<eT>& X)
75 {
76 arma_extra_debug_sigprint();
77
78 typedef typename get_pod_type<eT>::result T;
79
80 typedef std::complex<typename get_pod_type<eT>::result> out_eT;
81
82 arma_debug_check( (X.is_vec() == false), "roots(): given object must be a vector" );
83
84 if(X.is_finite() == false) { return false; }
85
86 // treat X as a column vector
87
88 const Col<eT> Y( const_cast<eT*>(X.memptr()), X.n_elem, false, false);
89
90 const T Y_max = (Y.is_empty() == false) ? T(max(abs(Y))) : T(0);
91
92 if(Y_max == T(0)) { out.set_size(1,0); return true; }
93
94 const uvec indices = find( Y / Y_max );
95
96 const uword n_tail_zeros = (indices.n_elem > 0) ? uword( (Y.n_elem-1) - indices[indices.n_elem-1] ) : uword(0);
97
98 const Col<eT> Z = Y.subvec( indices[0], indices[indices.n_elem-1] );
99
100 if(Z.n_elem >= uword(2))
101 {
102 Mat<eT> tmp;
103
104 if(Z.n_elem == uword(2))
105 {
106 tmp.set_size(1,1);
107
108 tmp[0] = -Z[1] / Z[0];
109 }
110 else
111 {
112 tmp = diagmat(ones< Col<eT> >(Z.n_elem - 2), -1);
113
114 tmp.row(0) = strans(-Z.subvec(1, Z.n_elem-1) / Z[0]);
115 }
116
117 Mat<out_eT> junk;
118
119 const bool status = auxlib::eig_gen(out, junk, false, tmp);
120
121 if(status == false) { return false; }
122
123 if(n_tail_zeros > 0)
124 {
125 out.resize(out.n_rows + n_tail_zeros, 1);
126 }
127 }
128 else
129 {
130 out.zeros(n_tail_zeros,1);
131 }
132
133 return true;
134 }

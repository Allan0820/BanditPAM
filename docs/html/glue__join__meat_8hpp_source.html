25glue_join_cols::apply_noalias(Mat<typename T1::elem_type>& out, const Proxy<T1>& A, const Proxy<T2>& B)
26 {
27 arma_extra_debug_sigprint();
28
29 const uword A_n_rows = A.get_n_rows();
30 const uword A_n_cols = A.get_n_cols();
31
32 const uword B_n_rows = B.get_n_rows();
33 const uword B_n_cols = B.get_n_cols();
34
35 arma_debug_check
36 (
37 ( (A_n_cols != B_n_cols) && ( (A_n_rows > 0) || (A_n_cols > 0) ) && ( (B_n_rows > 0) || (B_n_cols > 0) ) ),
38 "join_cols() / join_vert(): number of columns must be the same"
39 );
40
41 out.set_size( A_n_rows + B_n_rows, (std::max)(A_n_cols, B_n_cols) );
42
43 if( out.n_elem > 0 )
44 {
45 if(A.get_n_elem() > 0)
46 {
47 out.submat(0, 0, A_n_rows-1, out.n_cols-1) = A.Q;
48 }
49
50 if(B.get_n_elem() > 0)
51 {
52 out.submat(A_n_rows, 0, out.n_rows-1, out.n_cols-1) = B.Q;
53 }
54 }

337inline
338void
339glue_join_slices::apply(Cube<typename T1::elem_type>& out, const GlueCube<T1,T2,glue_join_slices>& X)
340 {
341 arma_extra_debug_sigprint();
342
343 typedef typename T1::elem_type eT;
344
345 const unwrap_cube<T1> A_tmp(X.A);
346 const unwrap_cube<T2> B_tmp(X.B);
347
348 const Cube<eT>& A = A_tmp.M;
349 const Cube<eT>& B = B_tmp.M;
350
351 if(A.n_elem == 0) { out = B; return; }
352 if(B.n_elem == 0) { out = A; return; }
353
354 arma_debug_check( ( (A.n_rows != B.n_rows) || (A.n_cols != B.n_cols) ), "join_slices(): size of slices must be the same" );
355
356 if( (&out != &A) && (&out != &B) )
357 {
358 out.set_size(A.n_rows, A.n_cols, A.n_slices + B.n_slices);
359
360 out.slices(0, A.n_slices-1 ) = A;
361 out.slices(A.n_slices, out.n_slices-1) = B;
362 }
363 else // we have aliasing
364 {
365 Cube<eT> C(A.n_rows, A.n_cols, A.n_slices + B.n_slices, arma_nozeros_indicator());
366
367 C.slices(0, A.n_slices-1) = A;
368 C.slices(A.n_slices, C.n_slices-1) = B;
369
370 out.steal_mem(C);
371 }

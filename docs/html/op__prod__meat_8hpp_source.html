22inline
23void
24op_prod::apply_noalias(Mat<eT>& out, const Mat<eT>& X, const uword dim)
25 {
26 arma_extra_debug_sigprint();
27
28 const uword X_n_rows = X.n_rows;
29 const uword X_n_cols = X.n_cols;
30
31 if(dim == 0) // traverse across rows (i.e. find the product in each column)
32 {
33 out.set_size(1, X_n_cols);
34
35 eT* out_mem = out.memptr();
36
37 for(uword col=0; col < X_n_cols; ++col)
38 {
39 out_mem[col] = arrayops::product(X.colptr(col), X_n_rows);
40 }
41 }
42 else // traverse across columns (i.e. find the product in each row)
43 {
44 out.ones(X_n_rows, 1);
45
46 eT* out_mem = out.memptr();
47
48 for(uword col=0; col < X_n_cols; ++col)
49 {
50 const eT* X_col_mem = X.colptr(col);
51
52 for(uword row=0; row < X_n_rows; ++row)
53 {
54 out_mem[row] *= X_col_mem[row];
55 }
56 }
57 }
58 }

93template<typename eT>
94inline
95eT
96op_prod::prod(const subview<eT>& X)
97 {
98 arma_extra_debug_sigprint();
99
100 eT val = eT(1);
101
102 const uword X_n_rows = X.n_rows;
103 const uword X_n_cols = X.n_cols;
104
105 if(X_n_rows == 1)
106 {
107 const Mat<eT>& A = X.m;
108
109 const uword start_row = X.aux_row1;
110 const uword start_col = X.aux_col1;
111
112 const uword end_col_p1 = start_col + X_n_cols;
113
114 uword i,j;
115 for(i=start_col, j=start_col+1; j < end_col_p1; i+=2, j+=2)
116 {
117 val *= A.at(start_row, i);
118 val *= A.at(start_row, j);
119 }
120
121 if(i < end_col_p1)
122 {
123 val *= A.at(start_row, i);
124 }
125 }
126 else
127 {
128 for(uword col=0; col < X_n_cols; ++col)
129 {
130 val *= arrayops::product( X.colptr(col), X_n_rows );
131 }
132 }
133
134 return val;
135 }

142op_prod::prod(const Base<typename T1::elem_type,T1>& X)
143 {
144 arma_extra_debug_sigprint();
145
146 typedef typename T1::elem_type eT;
147
148 const Proxy<T1> P(X.get_ref());
149
150 eT val = eT(1);
151
152 if(Proxy<T1>::use_at == false)
153 {
154 typedef typename Proxy<T1>::ea_type ea_type;
155
156 const ea_type A = P.get_ea();
157
158 const uword n_elem = P.get_n_elem();
159
160 uword i,j;
161 for(i=0, j=1; j < n_elem; i+=2, j+=2)
162 {
163 val *= A[i];
164 val *= A[j];
165 }
166
167 if(i < n_elem)
168 {
169 val *= A[i];
170 }
171 }
172 else
173 {
174 const uword n_rows = P.get_n_rows();
175 const uword n_cols = P.get_n_cols();
176
177 if(n_rows == 1)
178 {
179 uword i,j;
180 for(i=0, j=1; j < n_cols; i+=2, j+=2)
181 {
182 val *= P.at(0,i);
183 val *= P.at(0,j);
184 }
185
186 if(i < n_cols)
187 {
188 val *= P.at(0,i);
189 }
190 }
191 else
192 {
193 for(uword col=0; col < n_cols; ++col)
194 {
195 uword i,j;
196 for(i=0, j=1; j < n_rows; i+=2, j+=2)
197 {
198 val *= P.at(i,col);
199 val *= P.at(j,col);
200 }
201
202 if(i < n_rows)
203 {
204 val *= P.at(i,col);
205 }
206 }
207 }
208 }

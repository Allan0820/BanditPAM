47glue_polyval::apply(Mat<typename T1::elem_type>& out, const Glue<T1,T2,glue_polyval>& expr)
48 {
49 arma_extra_debug_sigprint();

53 const quasi_unwrap<T1> UP(expr.A);

56 const Mat<eT>& P = UP.M;

58
59 arma_debug_check( ((P.is_vec() == false) && (P.is_empty() == false)), "polyval(): argument P must be a vector" );
60
61 if(P.is_empty() || X.is_empty())
62 {
63 out.zeros(X.n_rows, X.n_cols);
64 return;
65 }
66
67 if(UP.is_alias(out) || UX.is_alias(out))
68 {
69 Mat<eT> tmp;

71 out.steal_mem(tmp);
72 }
73 else

76 }

23template<typename eT>
24inline
25void
26op_reshape::apply_unwrap(Mat<eT>& out, const Mat<eT>& A, const uword in_n_rows, const uword in_n_cols)
27 {
28 arma_extra_debug_sigprint();
29
30 const bool is_alias = (&out == &A);
31
32 const uword in_n_elem = in_n_rows * in_n_cols;
33
34 if(A.n_elem == in_n_elem)
35 {
36 if(is_alias == false)
37 {
38 out.set_size(in_n_rows, in_n_cols);
39 arrayops::copy( out.memptr(), A.memptr(), out.n_elem );
40 }
41 else // &out == &A, i.e. inplace reshape
42 {
43 out.set_size(in_n_rows, in_n_cols);
44 // set_size() doesn't destroy data as long as the number of elements in the matrix remains the same
45 }
46 }
47 else
48 {
49 const unwrap_check< Mat<eT> > B_tmp(A, is_alias);
50 const Mat<eT>& B = B_tmp.M;
51
52 const uword n_elem_to_copy = (std::min)(B.n_elem, in_n_elem);
53
54 out.set_size(in_n_rows, in_n_cols);
55
56 eT* out_mem = out.memptr();
57
58 arrayops::copy( out_mem, B.memptr(), n_elem_to_copy );
59
60 for(uword i=n_elem_to_copy; i<in_n_elem; ++i) { out_mem[i] = eT(0); }
61 }
62 }

66template<typename T1>
67inline
68void
69op_reshape::apply_proxy(Mat<typename T1::elem_type>& out, const Proxy<T1>& P, const uword in_n_rows, const uword in_n_cols)
70 {
71 arma_extra_debug_sigprint();
72
73 typedef typename T1::elem_type eT;
74
75 out.set_size(in_n_rows, in_n_cols);
76
77 eT* out_mem = out.memptr();
78
79 const uword in_n_elem = in_n_rows * in_n_cols;
80
81 if(P.get_n_elem() == in_n_elem)
82 {
83 if(Proxy<T1>::use_at == false)
84 {
85 typename Proxy<T1>::ea_type Pea = P.get_ea();
86
87 for(uword i=0; i < in_n_elem; ++i) { out_mem[i] = Pea[i]; }
88 }
89 else
90 {
91 const uword P_n_rows = P.get_n_rows();
92 const uword P_n_cols = P.get_n_cols();
93
94 for(uword col=0; col < P_n_cols; ++col)
95 {
96 uword i,j;
97
98 for(i=0, j=1; j < P_n_rows; i+=2, j+=2)
99 {
100 const eT tmp_i = P.at(i,col);
101 const eT tmp_j = P.at(j,col);
102
103 *out_mem = tmp_i; out_mem++;
104 *out_mem = tmp_j; out_mem++;
105 }
106
107 if(i < P_n_rows)
108 {
109 *out_mem = P.at(i,col); out_mem++;
110 }
111 }
112 }
113 }
114 else
115 {
116 const uword n_elem_to_copy = (std::min)(P.get_n_elem(), in_n_elem);
117
118 if(Proxy<T1>::use_at == false)
119 {
120 typename Proxy<T1>::ea_type Pea = P.get_ea();
121
122 for(uword i=0; i < n_elem_to_copy; ++i) { out_mem[i] = Pea[i]; }
123 }
124 else
125 {
126 uword i = 0;
127
128 const uword P_n_rows = P.get_n_rows();
129 const uword P_n_cols = P.get_n_cols();
130
131 for(uword col=0; col < P_n_cols; ++col)
132 for(uword row=0; row < P_n_rows; ++row)
133 {
134 if(i >= n_elem_to_copy) { goto nested_loop_end; }
135
136 out_mem[i] = P.at(row,col);
137
138 ++i;
139 }
140
141 nested_loop_end: ;
142 }
143
144 for(uword i=n_elem_to_copy; i < in_n_elem; ++i) { out_mem[i] = eT(0); }
145 }
146 }

150template<typename T1>
151inline
152void
153op_reshape::apply(Mat<typename T1::elem_type>& out, const Op<T1,op_reshape>& in)
154 {
155 arma_extra_debug_sigprint();
156
157 typedef typename T1::elem_type eT;
158
159 const uword in_n_rows = in.aux_uword_a;
160 const uword in_n_cols = in.aux_uword_b;
161
162 // allow detection of in-place reshape
163 if(is_Mat<T1>::value || is_Mat<typename Proxy<T1>::stored_type>::value)
164 {
165 const unwrap<T1> U(in.m);
166
167 op_reshape::apply_unwrap(out, U.M, in_n_rows, in_n_cols);
168 }
169 else
170 {
171 const Proxy<T1> P(in.m);
172
173 if(P.is_alias(out))
174 {
175 Mat<eT> tmp;
176
177 op_reshape::apply_proxy(tmp, P, in_n_rows, in_n_cols);
178
179 out.steal_mem(tmp);
180 }
181 else
182 {
183 op_reshape::apply_proxy(out, P, in_n_rows, in_n_cols);
184 }
185 }
186 }

190template<typename T1>
191inline
192void
193op_reshape::apply(Cube<typename T1::elem_type>& out, const OpCube<T1,op_reshape>& in)
194 {
195 arma_extra_debug_sigprint();
196
197 typedef typename T1::elem_type eT;
198
199 const unwrap_cube<T1> A_tmp(in.m);
200 const Cube<eT>& A = A_tmp.M;
201
202 const uword in_n_rows = in.aux_uword_a;
203 const uword in_n_cols = in.aux_uword_b;
204 const uword in_n_slices = in.aux_uword_c;
205
206 const uword in_n_elem = in_n_rows * in_n_cols * in_n_slices;
207
208 if(A.n_elem == in_n_elem)
209 {
210 if(&out != &A)
211 {
212 out.set_size(in_n_rows, in_n_cols, in_n_slices);
213 arrayops::copy( out.memptr(), A.memptr(), out.n_elem );
214 }
215 else // &out == &A, i.e. inplace resize
216 {
217 out.set_size(in_n_rows, in_n_cols, in_n_slices);
218 // set_size() doesn't destroy data as long as the number of elements in the cube remains the same
219 }
220 }
221 else
222 {
223 const unwrap_cube_check< Cube<eT> > B_tmp(A, out);
224 const Cube<eT>& B = B_tmp.M;
225
226 const uword n_elem_to_copy = (std::min)(B.n_elem, in_n_elem);
227
228 out.set_size(in_n_rows, in_n_cols, in_n_slices);
229
230 eT* out_mem = out.memptr();
231
232 arrayops::copy( out_mem, B.memptr(), n_elem_to_copy );
233
234 for(uword i=n_elem_to_copy; i < in_n_elem; ++i) { out_mem[i] = eT(0); }
235 }
236 }
